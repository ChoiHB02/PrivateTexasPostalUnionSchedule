## 원형 범위를 사용한 코드

원형 범위 안의 특정 타겟을 특정해 총알을 쏘는 코드
```C#
using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class FieldOfView : MonoBehaviour
{
    //캐릭터
    public GameObject agent;
    [Range(6, 60)]
    //원 드로잉에 사용할 선분의 매수
    public int lineCount; 
    //선분의 폭
    public float width;

    private LineRenderer LineDrawer;
    // 시야 영역의 반지름과 시야 각도
    public float viewRadius;

    // 원형 범위
    public float calcViewRadius => (float)(viewRadius + (GameController.instance.ingameData.atkRangeLv - 1) * 0.145);

    //시야각도
    [Range(0, 360)]
    public float viewAngle;

    //적 찾았는지 체크용 bool
    public bool FoundTarget;

    // 마스크 2종
    public LayerMask targetMask, obstacleMask;

    // Target mask에 ray hit된 transform을 보관하는 리스트
    public List<Transform> visibleTargets = new List<Transform>();

    //슈팅 관련 변수
    public float ShootTime;

    public float ShootTimeCalc => (float)ShootTime / (1 + GameController.instance.attackStatus.atkSpeed);

    public bool StopShoot = false;

    public static FieldOfView instance;

    private void Awake()
    {

        if (instance == null) instance = this;
    }

    void Start()
    {
        LineDrawer = GetComponent<LineRenderer>(); LineDrawer.loop = true;
        // 0.2초 간격으로 코루틴 호출
        StartCoroutine(FindTargetsWithDelay(0.2f));
        // 원형 드로우
        Draw(); 
        StartCoroutine(ShootWithDelay(ShootTime));
    }
    public void Draw() 
    {
        LineDrawer.positionCount = lineCount;
        LineDrawer.startWidth = width;
        // 원을 만들기 위한 하나의 선분 각도를 계산
        float theta = (2f * Mathf.PI) / lineCount; 
        float angle = 0;
        // 선분 개수만큼 원형으로 라인을 생성 => 원 
        for (int i = 0; i < lineCount; i++)
        {
            float x = calcViewRadius * Mathf.Cos(angle);
            float y = calcViewRadius * Mathf.Sin(angle);
            LineDrawer.SetPosition(i, new Vector3(x, y, 0));
            //switch 0 and y for 2D games
            angle += theta;
        }
    }
    IEnumerator FindTargetsWithDelay(float delay)
    {
        while (true)
        {
            yield return new WaitForSeconds(delay);
            FindVisibleTargets();
        }
    }

    void FindVisibleTargets()
    {
        visibleTargets.Clear();

        var vec = new Vector2(transform.position.x, transform.position.y);
        // viewRadius를 반지름으로 한 원 영역 내 targetMask 레이어인 콜라이더를 모두 가져옴
        Collider2D[] targetsInViewRadius = Physics2D.OverlapCircleAll(vec, calcViewRadius, targetMask);

        for (int i = 0; i < targetsInViewRadius.Length; i++)
        {
            Transform target = targetsInViewRadius[i].transform;
            Vector3 dirToTarget = (target.position - transform.position).normalized;

            // 플레이어와 up target이 이루는 각이 설정한 각도 내라면
            if (Vector3.Angle(transform.up, dirToTarget) < viewAngle / 2)
            {
                float dstToTarget = Vector3.Distance(transform.position, target.transform.position);

                // 타겟으로 가는 레이캐스트에 obstacleMask가 걸리지 않으면 visibleTargets에 Add
                if (!Physics2D.Raycast(transform.position, dirToTarget, dstToTarget, obstacleMask))
                {
                    visibleTargets.Add(target);
                }
            }
        }
        if (visibleTargets.Count >= 1)
        {
            FoundTarget = true;
        }
        else
        {
            FoundTarget = false;
        }
    }

    // y축 오일러 각을 3차원 방향 벡터로 변환한다.
    public Vector3 DirFromAngle(float angleDegrees, bool angleIsGlobal)
    {
        if (!angleIsGlobal)
        {
            angleDegrees += transform.eulerAngles.y;
        }
        return new Vector3(Mathf.Cos((-angleDegrees + 90) * Mathf.Deg2Rad), Mathf.Sin((-angleDegrees + 90) * Mathf.Deg2Rad),0);
    
    }
    
    // 이하 슈팅용 코드 원과 관련없음
    public float shortDis = 9999999999999999;
    public Transform _target;
    public IEnumerator ShootWithDelay(float delay)
    {
        while (true)
        {
            if (visibleTargets.Count >= 1)
            {
                foreach (var target in visibleTargets)
                {
                    try
                    {
                        float Distance = Vector3.Distance(gameObject.transform.position, target.transform.position);
                        //Debug.Log(Distance);
                        if (Distance < shortDis) // 위에서 잡은 기준으로 거리 재기
                        {
                           // Debug.Log("ok");
                            shortDis = Distance;
                            _target = target;
                        }
                    }
                    catch
                    {
                        visibleTargets.Clear();
                        break;
                        
                    }
                }
                try
                {
                    Shoot(_target);
                }
                catch
                {
                    visibleTargets.Clear();
                }
                shortDis = 99999999999;
            }
            if (StopShoot == true)
                break;
            yield return new WaitForSeconds(delay);
        }
    }
    public GameObject _bullet;
    public void Shoot(Transform target)
    {
        //Debug.Log("bullet");
        GameObject bullet = Instantiate(_bullet);
        bullet.SetActive(true);
        Bullet e = bullet.GetComponent<Bullet>();
        e.target = target.transform;
        
        bullet.transform.SetParent(agent.transform, false);
        bullet.transform.localPosition = this.transform.localPosition;
    }
    public void Stop()
    {
        StopShoot = true;
    }
    public void Startshoot()
    {
        StopShoot = false; StartCoroutine(ShootWithDelay(ShootTime));
    }
}
```
